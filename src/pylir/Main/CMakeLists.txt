  CommandLine.cpp
  CompilerInvocation.cpp
  DarwinToolchain.cpp

  MSVCToolchain.cpp
  MinGWToolchain.cpp
  PylirMain.cpp
  Toolchain.cpp
  Version.cpp
)

  PUBLIC
  Diagnostics
  
  ${llvm_options}
  
  MLIRPass
  PRIVATE
  CodeGen
  
  PylirHIRToPylirPy
  PylirHIRTransforms
  PylirLinker
  PylirOptimizer
  PylirToLLVMIR
  PylirMemDialect
  PylirMemTransforms
  PylirPyTransforms
  PylirPyToPylirMem
  PylirTransforms
  PylirLLVMPasses
  PylirExternalModels
  
  lldCOFF
  lldELF
  lldMachO
  lldMinGW
  lldWasm
  
  ${llvm_all}

  MLIRBytecodeReader

  MLIRControlFlowDialect
  MLIRDLTIDialect
  MLIRLLVMToLLVMIRTranslation
  
  ctre::ctre
